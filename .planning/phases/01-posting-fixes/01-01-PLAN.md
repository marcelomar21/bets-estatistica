/**

 * @param {string} text
 * @param {string|null|undefined} oddLabel - Label to use. If falsy, returns text unchanged.
 * @returns {string}

function enforceOddLabel(text, oddLabel) {
  if (!text || !oddLabel || oddLabel.trim() === '') return text || '';
